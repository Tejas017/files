import speech_recognition as sr
import re

DOCTOR_LIST = [
    "Dr. Michael Chen", "Dr. Sarah Williams",
    "Dr. James Rodriguez", "Dr. Emily Davis"
]

def listen_and_parse():
    recognizer = sr.Recognizer()
    with sr.Microphone() as source:
        recognizer.adjust_for_ambient_noise(source, duration=1)  # Better reliability
        print("Listening...")
        try:
            audio = recognizer.listen(source, timeout=5)
        except sr.WaitTimeoutError:
            return {"error": "No speech detected. Try again."}

    try:
        command = recognizer.recognize_google(audio)
        print("Recognized:", command)

        doctor = next((doc for doc in DOCTOR_LIST if doc.lower() in command.lower()), None)
        date_match = re.search(r"on (\d{4}-\d{2}-\d{2})", command)
        time_match = re.search(r"at (\d{1,2}:\d{2})", command)
        type_match = re.search(r"(checkup|consultation|follow[- ]?up|annual physical)", command, re.IGNORECASE)

        return {
            "spoken_text": command,
            "doctor": doctor,
            "date": date_match.group(1) if date_match else None,
            "time": time_match.group(1) if time_match else None,
            "type": type_match.group(1).title() if type_match else None
        }

    except sr.UnknownValueError:
        return {"error": "Could not understand audio."}
    except sr.RequestError as e:
        return {"error": f"API error: {str(e)}"}
    except Exception as e:
        return {"error": f"Unhandled error: {str(e)}"}
